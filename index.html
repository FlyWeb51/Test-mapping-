// County-level ACS profile for every county in a state — the "other counties" view. // Note: counties and congressional districts don't line up 1:1 (a county can span // several districts, and a district can contain parts of several counties), so this // is state-wide county browsing, not a per-district breakdown. const { cached, D30, CENSUS_KEY } = require("./_util"); const FIPS = {AL:"01",AK:"02",AZ:"04",AR:"05",CA:"06",CO:"08",CT:"09",DE:"10",FL:"12",GA:"13",HI:"15",ID:"16",IL:"17",IN:"18",IA:"19",KS:"20",KY:"21",LA:"22",ME:"23",MD:"24",MA:"25",MI:"26",MN:"27",MS:"28",MO:"29",MT:"30",NE:"31",NV:"32",NH:"33",NJ:"34",NM:"35",NY:"36",NC:"37",ND:"38",OH:"39",OK:"40",OR:"41",PA:"42",RI:"44",SC:"45",SD:"46",TN:"47",TX:"48",UT:"49",VT:"50",VA:"51",WA:"53",WV:"54",WI:"55",WY:"56"}; module.exports = async (req, res) => { try { const st = String(req.query.state || "").toUpperCase(); if (!FIPS[st]) return res.status(400).json({ error: "bad state" }); const data = await cached(`county:${st}`, D30, async () => { const vars = "NAME,DP05_0001E,DP03_0062E,DP02_0068PE,DP03_0128PE"; const key = CENSUS_KEY(); const url = `https://api.census.gov/data/2023/acs/acs5/profile?get=${vars}&for=county:*&in=state:${FIPS[st]}${key ? "&key=" + key : ""}`; const rows = await (await fetch(url)).json(); const header = rows[0]; const idx = k => header.indexOf(k); const n = x => (x == null || +x < 0) ? null : +x; const counties = rows.slice(1).map(r => ({ name: r[idx("NAME")], population: n(r[idx("DP05_0001E")])?.toLocaleString("en-US") ?? null, income: n(r[idx("DP03_0062E")]) != null ? "$" + n(r[idx("DP03_0062E")]).toLocaleString("en-US") : null, bachelors: n(r[idx("DP02_0068PE")]) != null ? n(r[idx("DP02_0068PE")]) + "%" : null, poverty: n(r[idx("DP03_0128PE")]) != null ? n(r[idx("DP03_0128PE")]) + "%" : null })).sort((a, b) => a.name.localeCompare(b.name)); return { state: st, counties }; }); res.json(data); } catch (e) { res.status(500).json({ error: e.message }); } };
